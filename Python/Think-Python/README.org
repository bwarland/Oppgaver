** Hvilke ord bruker ikke bokstaven "e"?

#+begin_src python
# l_in_word: str -> str*
# skriver ut alle bokstaver i et ord

def l_in_word(w):
    num_l=len(w)
    for l in range(0,num_l):
        print(w[l])

def has_no_e(word):
    num_letters=len(word)
    for index in range(0,num_letters):
        if word[index]=="e":
            return False
        else:
            return True
